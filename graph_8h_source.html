34 /// @class Graph
35 ///
36 /// @brief A Graph represents the relationship between a variety of nodes. It
37 /// contains a collection of nodes that are linked together by their
38 /// edges.
39 ///
40 /// Nodes in a graph may connect their input edges to the output edges of other
41 /// nodes or they may simply specify a default value.
42 ///
43 /// Graphs cannot execute nodes directly. To have nodes executed, a Graph object
44 /// must be passed to a GraphState object which tracks the mutable state of the
45 /// data in the graph. This also allows you to reuse one Graph with multiple
46 /// game entities or actors. Once a GraphState has been initialized, events
47 /// broadcast to the nodes in the graph will trigger the nodes in the graph.
48 class Graph {
49  public:
50  /// @brief Construct a graph.
51  ///
52  /// @param[in] graph_name An arbitrary string used to identify this graph.
53  /// Typically this is just the filename of the file that loaded
54  /// this Graph, but this is not required. This string shows up in
55  /// the error logs if there are errors with the graph.
56  explicit Graph(const std::string& graph_name)
57  : graph_name_(graph_name),
58  nodes_(),
59  sorted_nodes_(),
60  nodes_finalized_(false) {}
61 
62  /// @brief Destructor for a BaseNode.
63  ~Graph();
64 
65  /// @brief Construct a graph.
66  ///
67  /// @return Returns the name of this Graph.
68  const std::string& graph_name() const { return graph_name_; }
69 
70  /// @brief Add a new node with the given signature to the Graph.
71  ///
72  /// @param[in] signature The signature of the new node to add to the graph.
73  ///
74  /// @return Returns the node that was just added so that its input edges can
75  /// be set up.
76  Node* AddNode(const NodeSignature* signature);
77 
78  /// @brief FinalizeNodes should be called once, after all nodes have been
79  /// added to the Graph, and before setting default input edge values.
80  ///
81  /// Setting up a graph comes in two distinct steps. First the nodes must be
82  /// added and the connections between the nodes defined. Once all Nodes have
83  /// been added, FinalizeNodes must be called. After FinalizeNodes has been
84  /// called memory has been allocated in which default input values may be
85  /// stored. It is at this point that you may call SetDefaultValue to set the
86  /// default input values for nodes.
87  ///
88  /// Call once after graph is set up. No more node additions or modifications
89  /// may be performed after this is called.
90  ///
91  /// @return Returns true if successful. Otherwise an error is logged and false
92  // is returned.
93  bool FinalizeNodes();
94 
95  /// @brief Returns true if FinalizeNodes has been called.
96  ///
97  /// @return Returns true if FinalizeNodes has been called.
98  bool nodes_finalized() const { return nodes_finalized_; }
99 
100  /// @brief Set the default input edge argument for a node.
101  ///
102  /// Default values may only be set once the nodes are finalized using
103  /// FinalizeNodes.
104  ///
105  /// @param[in] node_index The index of the node.
106  /// @param[in] edge_index The index of the edge on the node.
107  /// @param[in] value The value to set the given input edge to. The templated
108  /// type of this value must match the expected type, defined in the
109  /// NodeSignature.
110  template <typename EdgeType>
111  void SetDefaultValue(unsigned int node_index, unsigned int edge_index,
112  const EdgeType& value) {
113  assert(nodes_finalized_);
114  if (node_index >= nodes_.size()) {
115  CallLogFunc(
116  "%s: Attempting to assign a default value on node %d when graph only "
117  "has %d nodes.",
118  graph_name_.c_str(), node_index, static_cast<int>(nodes_.size()));
119  return;
120  }
121  Node& node = nodes_[node_index];
122  const NodeSignature* signature = node.signature();
123  if (edge_index >= node.input_edges().size()) {
124  CallLogFunc(
125  "%s: Attempting to assign a default value to node %i (%s:%s), edge "
126  "%d when node only has %d input edges.",
127  graph_name_.c_str(), node_index, signature->module_name()->c_str(),
128  signature->node_name().c_str(), edge_index,
129  static_cast<int>(node.input_edges().size()));
130  return;
131  }
132  const Type* type = TypeRegistry<EdgeType>::GetType();
133  const Type* expected_type = signature->input_parameters()[edge_index].type;
134  if (type != expected_type) {
135  CallLogFunc(
136  "%s: Attempting to assign a default value of the incorrect type to "
137  "node %d (%s:%s), edge %d. Edge got type \"%s\" when it expects type "
138  "\"%s\".",
139  graph_name_.c_str(), node_index, signature->module_name()->c_str(),
140  signature->node_name().c_str(), edge_index, type->name,
141  expected_type->name);
142  return;
143  }
144  InputEdge& input_edge = node.input_edges()[edge_index];
145  EdgeType* default_object =
146  input_buffer_.GetObject<EdgeType>(input_edge.data_offset());
147  *default_object = value;
148  }
