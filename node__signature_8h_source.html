46 /// @class NodeSignature
47 ///
48 /// @brief A NodeSignature is a class that represents the various input and
49 /// output edges of a node, as well as other metadata about the node itself.
50 ///
51 /// A NodeSignature defines how many input and ouput edges a Node in a graph is
52 /// expected to have, and what types those edges are. Additionally,
53 /// NodeSignatures have a constructor callback which constructs an instance the
54 /// class that implements BaseNode. The default constructor is used by default,
55 /// but you may supply your own constructor callback if you need to do any
56 /// special set up before creating an instance of your class.
57 ///
58 /// @note You do not need to construct a NodeSignature directly; they are
59 /// created by the ModuleRegistry when a new node is registered with it.
60 class NodeSignature {
61  public:
62  /// @brief Construct a NodeSignature.
63  ///
64  /// param[in] module_name The name of the module of the node type that this
65  /// NodeSignature represents.
66  ///
67  /// param[in] node_name The name of the node that this NodeSignature
68  /// represents.
69  ///
70  /// param[in] constructor A function that returns a new instance of the node
71  /// type that this NodeSignature represents.
72  ///
73  /// param[in] destructor A function that deletes an existing instance of the
74  /// node type that this NodeSignature represents.
75  NodeSignature(const std::string* module_name, const std::string& node_name,
76  const NodeConstructor& constructor,
77  const NodeDestructor& destructor)
78  : module_name_(module_name),
79  node_name_(node_name),
80  constructor_(constructor),
81  destructor_(destructor) {}

94  /// @brief Adds an input edge to a node signature.
95  ///
96  /// Call this function once for each input edge parameter on the node,
97  /// supplying the type of the parameter as the template parameter each
98  /// time. Typical usage would look like this:
99  ///
100  /// ~~~{.cpp}
101  /// class EqualsNode : public BaseNode {
102  /// public:
103  /// static void OnRegister(NodeSignature* node_sig) {
104  /// // Adds two inut arguments of type std::string and one output
105  /// // argument of type bool.
106  /// node_sig->AddInput<std::string>();
107  /// node_sig->AddInput<std::string>();
108  /// node_sig->AddOutput<bool>();
109  /// }
110  /// }
111  /// ~~~
112  template <typename EdgeType>
113  void AddInput() {
114  input_types_.push_back(TypeRegistry<EdgeType>::GetType());
115  }
116 
117  /// @brief Adds an output edge to a node signature.
118  ///
119  /// Call this function once for each output edge parameter on the node,
120  /// supplying the type of the parameter as the template parameter each
121  /// time. Typical usage would look like this:
122  ///
123  /// ~~~{.cpp}
124  /// class EqualsNode : public BaseNode {
125  /// public:
126  /// static void OnRegister(NodeSignature* node_sig) {
127  /// // Adds two inut arguments of type std::string and one output
128  /// // argument of type bool.
129  /// node_sig->AddInput<std::string>();
130  /// node_sig->AddInput<std::string>();
131  /// node_sig->AddOutput<bool>();
132  /// }
133  /// }
134  /// ~~~
135  template <typename EdgeType>
136  void AddOutput() {
137  output_types_.push_back(TypeRegistry<EdgeType>::GetType());
138  }
139 
140  /// @brief Adds a listener to a node signature.
141  ///
142  /// A node can have any number of listeners associated with it. Each listener
143  /// may listen for one and exactly one EventId. A NodeEventListener must be
144  /// registered with a NodeEventBroadcaster to recieve events.
145  ///
146  /// Typical usage would look like this:
147  ///
148  /// ~~~{.cpp}
149  /// class ExampleNode : public breadboard::BaseNode {
150  /// public:
151  /// static void OnRegister(NodeSignature* signature) {
152  /// signature->AddInput<GameEntity>();
153  /// // The NodeEventListener is created and tracked internally here.
154  /// signature->AddListener(kMyEventId);
155  /// }
156  /// };
157  /// ~~~
158  ///
159  /// @param[in] event_id The event id that this listener should listen for.
160  void AddListener(const EventId event_id) {
161  event_listeners_.push_back(event_id);
162  }
