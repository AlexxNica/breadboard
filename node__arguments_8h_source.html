33 /// @class NodeArguments
34 ///
35 /// @brief NodeArguments is a class that represents the state of all input and
36 /// output edges connected to a node.
37 ///
38 /// Functionality is provided to get inputs and set outputs. Additionally, this
39 /// contains the list of listeners on a given node so that you can connect a
40 /// listener to a broadcaster.
41 class NodeArguments {
42  public:
43  /// @cond BREADBOARD_INTERNAL
44  /// @brief Construct a NodeArguments object for the given node.
45  ///
46  /// @note For internal use only.
47  ///
48  /// @param[in] node The node these arguments refer to.
49  ///
50  /// @param[in] nodes The list of nodes on the graph.
51  ///
52  /// @param[in] input_memory The MemoryBuffer for the input edges.
53  ///
54  /// @param[in] output_memory The MemoryBuffer for the output edges.
55  ///
56  /// @param[in] timestamp The current timestamp.
57  NodeArguments(const Node* node, const std::vector<Node>* nodes,
58  MemoryBuffer* input_memory, MemoryBuffer* output_memory,
59  Timestamp timestamp)
60  : node_(node),
61  nodes_(nodes),
62  input_memory_(input_memory),
63  output_memory_(output_memory),
64  timestamp_(timestamp) {}
65  /// @endcond BREADBOARD_INTERNAL
66 
67  /// @brief Returns the value of this input edge.
68  ///
69  /// Each node has number of typed inputs (specified by the
70  /// NodeSignature). This function allows you to access the data held in any
71  /// one of those inputs. Typical usage would look like this:
72  ///
73  /// ~~~{.cpp}
74  /// Foo* foo = args->GetInput<Foo>(0);
75  /// Bar* bar = args->GetInput<Bar>(1);
76  /// ~~~
77  ///
78  /// Each argument is accessed by index, and the types must match the types
79  /// declared in the NodeSignature. Care must be taken when modifying input
80  /// values as they will *not* be marked dirty. Since the execution order of
81  /// nodes is not defined, input values may change between calls to Execute on
82  /// different nodes.
83  ///
84  /// @param argument_index The index of the input edge. Node that the template
85  /// argument of this index must match the one specified in the NodeSignature.
86  ///
87  /// @return A pointer to the value of the given input edge.
88  template <typename EdgeType>
89  EdgeType* GetInput(size_t argument_index) const {
90  VerifyInputPreconditions(argument_index, TypeRegistry<EdgeType>::GetType());
91 
92  const InputEdge& input_edge = node_->input_edges()[argument_index];
93  if (input_edge.connected()) {
94  const OutputEdgeTarget& target_edge = input_edge.target();
95  const OutputEdge& output_edge = target_edge.GetTargetEdge(nodes_);
96  return output_memory_->GetObject<EdgeType>(output_edge.data_offset());
97  } else {
98  return input_memory_->GetObject<EdgeType>(input_edge.data_offset());
99  }
100  }
101 
102  /// @brief Returns true if the given input argument index has been modified
103  /// since the last execution of this graph.
104  ///
105  /// @return True if the given input argument index has been modified since the
106  /// last execution of this graph.
107  bool IsInputDirty(size_t argument_index) const;
108 
109  /// @brief Returns true if the given listener index has been signaled by a
110  /// broadcaster since the last execution of this graph.
111  ///
112  /// @return True if the given listener index has been signaled by a
113  /// broadcaster since the last execution of this graph.
114  bool IsListenerDirty(size_t listener_index) const;
115 
116  /// @brief Sets the value of the specified output edge.
117  ///
118  /// Each node has number of typed outputs (specified by the NodeSignature).
119  /// This function allows you to pass data through an output edge so that it is
120  /// accessible to any input edge that is connected to it.
121  ///
122  /// Typical usage would look like this:
123  ///
124  /// ~~~{.cpp}
125  /// Foo foo = ...
126  /// Bar bar = ...
127  /// outputs.SetOutput(0, foo);
128  /// outputs.SetOutput(1, bar);
129  /// ~~~
130  ///
131  /// Each output is accessed by index, and the types must match the types
132  /// declared in the NodeSignature. When you set a value that edge is marked
133  /// dirty, meaning that any node that relies on it will re-run its Execute
134  /// function. If an edge is not connected to any inputs, the value is
135  /// discarded and this function call does nothing.
136  ///
137  /// @param argument_index The index of the input edge. Node that the template
138  /// argument of this index must match the one specified in the NodeSignature.
139  ///
140  /// @param value The value to set this output edge to.
141  template <typename EdgeType>
142  void SetOutput(size_t argument_index, const EdgeType& value) {
143  VerifyOutputPreconditions(argument_index,
144  TypeRegistry<EdgeType>::GetType());
145 
146  const OutputEdge& output_edge = node_->output_edges()[argument_index];
147  if (!output_edge.connected()) {
148  // Nothing is consuming this output, so no need to store it.
149  return;
150  }
151 
152  // Mark that this value has changed.
153  Timestamp* timestamp =
154  output_memory_->GetObject<Timestamp>(output_edge.timestamp_offset());
155  *timestamp = timestamp_;
156 
157  EdgeType* data =
158  output_memory_->GetObject<EdgeType>(output_edge.data_offset());
159  *data = value;
160  }
161 
162  /// @brief Marks an output edge as dirty without updating its value.
163  ///
164  /// This is a special overload of SetOutput for void output types, which are
165  /// typically used just to send a signal to another node but which carry no
166  /// data. With this version of the function you do not need to supply any
167  /// value. Typical usage would look like this:
168  ///
169  /// ~~~{.cpp}
170  /// outputs.SetOutput(0);
171  /// outputs.SetOutput(1);
172  /// ~~~
173  ///
174  /// @param argument_index The index of the input edge. Node that the template
175  /// argument of this index must be `void`.
176  void SetOutput(size_t argument_index) {
177  VerifyOutputPreconditions(argument_index, TypeRegistry<void>::GetType());
178 
179  const OutputEdge& output_edge = node_->output_edges()[argument_index];
180  if (!output_edge.connected()) {
181  // Nothing is consuming this output, so no need to store it.
182  return;
183  }
184 
185  // Mark that this value has changed.
186  Timestamp* timestamp =
187  output_memory_->GetObject<Timestamp>(output_edge.timestamp_offset());
188  *timestamp = timestamp_;
189  }
190 
191  /// @brief Binds the given broadcasater to the listener at the given index.
192  ///
193  /// An NodeEventListener cannot respond to events until it has been bound to a
194  /// NodeEventBroadcaster. Use this function to bind a broadcaster to a
195  /// listener.
196  ///
197  /// @param listener_index The index of the listener.
198  ///
199  /// @param broadcaster The broadcaster to bind the given listener to.
200  void BindBroadcaster(size_t listener_index,
201  NodeEventBroadcaster* broadcaster) {
202  NodeEventListener* listener = output_memory_->GetObject<NodeEventListener>(
203  node_->listener_offsets()[listener_index]);
204  broadcaster->RegisterListener(listener);
205  }
206 
207  private:
208  // Check to make sure the argument index is in range and the type being
209  // retrieved is the type expected.
210  void VerifyInputPreconditions(size_t argument_index,
211  const Type* requested_type) const;
212 
213  // Check to make sure the argument index is in range and the type being set is
214  // the type expected.
215  void VerifyOutputPreconditions(size_t argument_index,
216  const Type* requested_type) const;
217 
218  void VerifyListenerPreconditions(size_t listener_index) const;
219 
220  const Node* node_;
221  const std::vector<Node>* nodes_;
222  MemoryBuffer* input_memory_;
223  MemoryBuffer* output_memory_;
224  Timestamp timestamp_;
225 };
